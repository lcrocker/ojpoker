checkTwoPair() {
  if (cards.length < 4) {
    twoPair = TFALSE;
    return;
  }
  assert(sorted.isTrue());
  assert(quads.isFalse());
  assert(fullHouse.isFalse());
  assert(trips.isFalse());

  // AABBC
  if (ranks[0] == ranks[1] && ranks[2] == ranks[3]) {
    twoPair = TTRUE;
    return;
  }
  if (cards.length < 5) {
    twoPair = TFALSE;
    return;
  }
  // ABBCC
  if (ranks[1] == ranks[2] && ranks[3] == ranks[4]) {
    ranks[4] = ranks[0];
    ranks[0] = ranks[1];
    ranks[2] = ranks[3];

    sorted = TFALSE;
    twoPair = TTRUE;
    return;
  }
  // AABCC
  if (ranks[0] == ranks[1] && ranks[3] == ranks[4]) {
    ranks[4] = ranks[2];
    ranks[2] = ranks[3];

    sorted = TFALSE;
    twoPair = TTRUE;
    return;
  }
  twoPair = TFALSE;
}
