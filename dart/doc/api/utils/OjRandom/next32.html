int next32()

Generate a random 32-bit integer. While Dart uses 64-bit integers, it can be compiled into JavaScript, which doesn't play nice with 64-bit ints, so we stick to 32 here.

Implementation

static int next32() {
  if (!_seeded) {
    _seeded = true;
    _s[0] = DateTime.now().millisecondsSinceEpoch;
    _s[1] = _s[0] + 1;
    _s[2] = _s[0] + 2;
    _s[3] = _s[0] + 3;
  }

  int r = (_s[0] + _s[3]) & 0xFFFFFFFF;
  final int result = ((r << 7) | (r >> 25)) + _s[0];
  final int t = _s[1] << 9;

  _s[2] ^= _s[0];
  _s[3] ^= _s[1];
  _s[1] ^= _s[2];
  _s[0] ^= _s[3];

  _s[2] ^= t;
  _s[3] = (_s[3] << 11) | ((_s[3] & 0xFFFFFFFF) >> 21);

  return result & 0xFFFFFFFF;
}
